// Função para baixar o PDF
document.getElementById('download-pdf').addEventListener('click', function() {
    const { jsPDF } = window.jspdf;
    const img = new Image();
    img.src = 'https://i.ibb.co/7XXLP89/Noshigami-Okaeshi-LOGO-NOVO-36-5cm-x-16cm.jpg';
    
    img.onload = function() {
        const imgWidthMM = 365; // Largura da imagem em milímetros
        const imgHeightMM = 160; // Altura da imagem em milímetros
        const fontSizeMM = 25 * 0.352778; // Tamanho fixo da fonte em pixels convertido para milímetros

        const doc = new jsPDF('l', 'mm', [imgWidthMM, imgHeightMM]);

        doc.addImage(img, 'PNG', 0, 0, imgWidthMM, imgHeightMM);

        const textoPosition = $('#texto-fixo').position();
        const textoX = textoPosition.left;
        const textoY = textoPosition.top;

        // PPI de 133: conversão de pixels para milímetros
        const pixelToMM = 25.4 / 133; // 1 pixel ≈ 0.19082 mm

        const pdfX = textoX * pixelToMM; // Converte X para milímetros
        const pdfY = textoY * pixelToMM; // Converte Y para milímetros

        doc.setFontSize(fontSizeMM); // Define o tamanho fixo da fonte no PDF
        doc.text(previewTexto.textContent, pdfX, pdfY, {
            angle: 0, // Sem rotação
            align: "center"
        });

        doc.save('Noshigami.pdf');
    };
});

// Função para gerar o arquivo DOCX
document.getElementById('gerar-docx').addEventListener('click', function() {
    const { Document, Packer, Paragraph, TextRun, PageSize } = docx;

    // Definindo largura e altura em pontos (1 ponto = 1/72 de polegada)
    const largura = 918; // 36,5 cm em pontos 
    const altura = 381; // 16 cm em pontos 

    const doc = new Document({
        creator: 'Seu Nome ou Aplicativo',
        description: 'Documento com tamanho personalizado',
        sections: [{
            properties: {
                pageSize: {
                  width: { size: largura, type: 'dxa' }, 
                  height: { size: altura, type: 'dxa' }
                },
                page: {
                    margin: {
                        top: 0,
                        bottom: 0,
                        left: 0,
                        right: 0,
                    },
                },
            },
            children: [
                new Paragraph({
                    children: [
                        new TextRun(Nome do Falecido: ${nomeFalecido.value}),
                        new TextRun({
                            text: "\n",
                        }),
                        new TextRun(Nome do Falecido em Japonês: ${nomeFalecidoJapones.value}),
                        new TextRun({
                            text: "\n",
                        }),
                        new TextRun(Nome da Família: ${nomeFamilia.value}),
                        new TextRun({
                            text: "\n",
                        }),
                        new TextRun(Nome da Família em Japonês: ${nomeFamiliaJapones.value}),
                        new TextRun({
                            text: "\n",
                        }),
                        new TextRun(Missa Numeral: ${periodonumeral.value}),
                        new TextRun({
                            text: "\n",
                        }),
                        new TextRun(Missa de: ${periodo.value}),
                        new TextRun({
                            text: "\n",
                        }),
                        new TextRun(Relação: ${relacao.value})
                    ],
                }),
            ],
        }],
    });

    Packer.toBlob(doc).then(blob => {
        console.log(blob);
        saveAs(blob, "documento.docx"); // Usando FileSaver.js
        console.log("Document created successfully");
    }).catch(err => {
        console.error('Erro ao gerar o arquivo DOCX:', err);
    });
});
